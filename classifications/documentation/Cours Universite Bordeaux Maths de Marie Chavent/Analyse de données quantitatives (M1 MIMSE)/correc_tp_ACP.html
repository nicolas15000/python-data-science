#############ACP TP1: Analyse du nuage des individus ######Importation du tableau de données###### X<-read.table("eaux.txt", header=TRUE, row.names=1,sep=";") n<-nrow(X) p<-ncol(X) colnames<-colnames(X) rownames<-rownames(X) ######Calcul de la matrice centrée réduite avec la fonction scale ()###### moy<-apply(X,2,mean) ecartype<-sd(X) # mais numérateur n-1 donc ecartype<-sd(X)*sqrt((n-1)/n) #pour obtenir l'écart-type corrigé! ?scale Z<-scale(X,center=moy,scale=ecartype) apply(Z,2,mean) #verification de la nullité des moyennes apply(Z,2,sd)*sqrt((n-1)/n) #verification des ecart-types à 1 ######Calcul de la matrice des corrélations###### #N<-matrix(0,n,n) N<-diag(rep(1/n,n)) R<-t(Z)%*%N%*%Z ######Décomposition spéctrale avec la fonction eigen()###### e<-eigen(R) e$values e$vectors t(e$vectors)%*%e$vectors #les vecteurs propres sont bien orhonormés sum(e$values) ######Matrice des coordonnées factorielles des individus (des scores)###### Psi<-Z%*%e$vectors apply(Psi,2,mean) #verification de la nullité des moyennes apply(Psi,2,var)*(n-1)/n e$values ######Représentation graphique sur les plans factoriels des individus###### par(mfrow=c(1,2)) plot(Psi[,1:2]) text(Psi[,1:2],rownames(X)) plot(Psi[,2:3]) text(Psi[,2:3],rownames(X)) ######Inerties###### sum(e$values) #Inertie totale du nuage des 5 eaux minérales e$value/p*100
